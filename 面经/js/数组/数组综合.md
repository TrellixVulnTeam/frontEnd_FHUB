# 数组综合

## 1

``` javascript
执行以下程序，下列说法中，正确的是（）
var arr = new Array(3); ...①
arr[0] = 1;
arr.b  = 0;
console.log(arr.length); ...②
arr.forEach(value=>{
        console.log(value); ...③
})
for(var i in arr){
      console.log(arr[i]); ...④

}

/*
	var arr;
	arr = [],长度为3
	arr[0] = 1
	arr.b = 0; 给数组绑定了属性b，值为0；
	console.log(arr.length) //输出3，未赋值的部分为undefined
	arr.forEach(value =>{
		console.log(value) // 打印数组每项的值 1 undefined undefined
	})
	for(var i in arr){
		console.log(arr[i]) //打印数组对象的属性 3(数组长度属性的值) 0(数组属性b的值)
	}
*/
```
